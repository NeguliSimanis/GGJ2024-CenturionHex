142 {
143 return (1 - t) * (1 - t) * (1 - t) * p0 + 3 * (1 - t) * (1 - t) * t * c0 + 3 * (1 - t) * t * t * c1 + t * t * t * p1;

146 private Vector3 dbezier (Vector3 p0, Vector3 c0, Vector3 c1, Vector3 p1, float t)
147 {
148 Vector3 dir = 3 * (1 - t) * (1 - t) * (c0 - p0) + 6 * (1 - t) * t * (c1 - c0) + 3 * t * t * (p1 - c1);
149 if (dir.magnitude < 0.001) {
150 return 6 * (1 - t) * (c1 - 2 * c0 + p0) + 6 * t * (p1 - 2 * c1 + c0);
