28 /// <summary>
29 /// Returns a random integer with a negative binomial distribution and the given values of r and p.
30 /// For integer values of r, it gives results as if attempts are made with probability p of success until there are r failures.
31 /// The number of successes is returned.
32 /// </summary>
33 /// <returns>A random integer with a negative binomial distribution and the given values of r and p.</returns>
34 /// <param name="r">The number of failures.</param>
35 /// <param name="p">The probability of success.</param>
36 public static int fromRAndP (float r, float p)
37 {
38 float pmf = Mathf.Pow (1 - p, r);
39 float rand = Random.value;
40 int k = 0;
41 while (k < 256) {
42 if (rand < pmf) {
43 return k;
44 } else {
45 rand -= pmf;
46 ++k;
47 pmf *= (k + r - 1) * p / k;
48 }
49 }
50 return k;
51 }

53 /// <summary>
54 /// Returns a random integer with a Poisson distribution and the given value of lambda.
55 /// This distribution is what you'd expect for the number of heads if there are lots of coins and they're each very unlikely to land on heads.
56 /// It's also the limit of negative binomial distribution for the smallest standard deviation and a given mean.
57 /// </summary>
