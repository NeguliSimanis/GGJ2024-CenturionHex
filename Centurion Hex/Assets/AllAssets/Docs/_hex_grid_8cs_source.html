212 private HexPosition getMouseHex ()
213 {
214 Ray ray = Camera.main.ScreenPointToRay (Input.mousePosition);
215 RaycastHit[] hits = Physics.RaycastAll (ray);
216 if (hits.Length == 0) {
217 return null;
218 } else {
219 float minDist = float.PositiveInfinity;
220 int min = 0;
221 for (int i = 0; i < hits.Length; ++i) {
222 if (hits [i].distance < minDist) {
223 minDist = hits [i].distance;
224 min = i;
225 }
226 }
227 return (new HexPosition (hits [min].point));
