15 /// <summary>
16 /// SimpleRNG is a simple random number generator based on
17 /// George Marsaglia's MWC (multiply with carry) generator.
18 /// Although it is very simple, it passes Marsaglia's DIEHARD
19 /// series of random number generator tests.
20 ///
21 /// Written by John D. Cook
22 /// http://www.johndcook.com
23 /// </summary>
24 public class SimpleRNG

27 private static uint m_z;

29 static SimpleRNG ()

31 // These values are not magical, just the default values Marsaglia used.
32 // Any pair of unsigned integers should be fine.
33 m_w = 521288629;

62 // Produce a uniform random sample from the open interval (0, 1).
63 // The method will not return either end point.
64 public static double GetUniform ()
65 {
66 // 0 <= u < 2^32
67 uint u = GetUint ();
68 // The magic number below is 1/(2^32 + 2).
69 // The result is strictly between 0 and 1.
70 return (u + 1.0) * 2.328306435454494e-10;
71 }
72
73 // This is the heart of the generator.
74 // It uses George Marsaglia's MWC algorithm to produce an unsigned integer.
75 // See http://www.bobwheeler.com/statistics/Password/MarsagliaPost.txt
76 private static uint GetUint ()
77 {
78 m_z = 36969 * (m_z & 65535) + (m_z >> 16);
79 m_w = 18000 * (m_w & 65535) + (m_w >> 16);
80 return (m_z << 16) + m_w;
81 }

83 // Get normal (Gaussian) random sample with mean 0 and standard deviation 1
84 public static double GetNormal ()
85 {
86 // Use Box-Muller algorithm
87 double u1 = GetUniform ();
88 double u2 = GetUniform ();
89 double r = Math.Sqrt (-2.0 * Math.Log (u1));
90 double theta = 2.0 * Math.PI * u2;
91 return r * Math.Sin (theta);
92 }

94 // Get normal (Gaussian) random sample with specified mean and standard deviation
95 public static double GetNormal (double mean, double standardDeviation)
96 {
97 if (standardDeviation <= 0.0) {
98 string msg = string.Format ("Shape must be positive. Received {0}.", standardDeviation);
99 throw new ArgumentOutOfRangeException (msg);
100 }
101 return mean + standardDeviation * GetNormal ();
102 }

120 public static double GetGamma (double shape, double scale)
121 {
122 // Implementation based on "A Simple Method for Generating Gamma Variables"
123 // by George Marsaglia and Wai Wan Tsang. ACM Transactions on Mathematical Software
124 // Vol 26, No 3, September 2000, pages 363-372.
125
126 double d, c, x, xsquared, v, u;
127
128 if (shape >= 1.0) {
129 d = shape - 1.0 / 3.0;
130 c = 1.0 / Math.Sqrt (9.0 * d);
131 for (;;) {
132 do {
133 x = GetNormal ();
134 v = 1.0 + c * x;
135 } while (v <= 0.0);
136 v = v * v * v;
137 u = GetUniform ();
138 xsquared = x * x;
139 if (u < 1.0 - .0331 * xsquared * xsquared || Math.Log (u) < 0.5 * xsquared + d * (1.0 - v + Math.Log (v)))
140 return scale * d * v;
141 }
142 } else if (shape <= 0.0) {
143 string msg = string.Format ("Shape must be positive. Received {0}.", shape);
144 throw new ArgumentOutOfRangeException (msg);
145 } else {
146 double g = GetGamma (shape + 1.0, 1.0);
147 double w = GetUniform ();
148 return scale * g * Math.Pow (w, 1.0 / shape);
149 }
150 }

152 public static double GetChiSquare (double degreesOfFreedom)
153 {
154 // A chi squared distribution with n degrees of freedom
155 // is a gamma distribution with shape n/2 and scale 2.
156 return GetGamma (0.5 * degreesOfFreedom, 2.0);
157 }

159 public static double GetInverseGamma (double shape, double scale)
160 {
161 // If X is gamma(shape, scale) then
162 // 1/Y is inverse gamma(shape, 1/scale)
163 return 1.0 / GetGamma (shape, 1.0 / scale);

166 public static double GetWeibull (double shape, double scale)
167 {
168 if (shape <= 0.0 || scale <= 0.0) {
169 string msg = string.Format ("Shape and scale parameters must be positive. Recieved shape {0} and scale{1}.", shape, scale);
170 throw new ArgumentOutOfRangeException (msg);

172 return scale * Math.Pow (-Math.Log (GetUniform ()), 1.0 / shape);

175 public static double GetCauchy (double median, double scale)
176 {
177 if (scale <= 0) {
178 string msg = string.Format ("Scale must be positive. Received {0}.", scale);
179 throw new ArgumentException (msg);
180 }
181
182 double p = GetUniform ();
183
184 // Apply inverse of the Cauchy distribution function to a uniform
185 return median + scale * Math.Tan (Math.PI * (p - 0.5));
186 }

188 public static double GetStudentT (double degreesOfFreedom)
189 {
190 if (degreesOfFreedom <= 0) {
191 string msg = string.Format ("Degrees of freedom must be positive. Received {0}.", degreesOfFreedom);
192 throw new ArgumentException (msg);
193 }
194
195 // See Seminumerical Algorithms by Knuth
196 double y1 = GetNormal ();
197 double y2 = GetChiSquare (degreesOfFreedom);
198 return y1 / Math.Sqrt (y2 / degreesOfFreedom);
199 }

201 // The Laplace distribution is also known as the double exponential distribution.
202 public static double GetLaplace (double mean, double scale)
203 {
204 double u = GetUniform ();
205 return (u < 0.5) ?
206 mean + scale * Math.Log (2.0 * u) :
207 mean - scale * Math.Log (2 * (1 - u));
208 }

210 public static double GetLogNormal (double mu, double sigma)
211 {
212 return Math.Exp (GetNormal (mu, sigma));

215 public static double GetBeta (double a, double b)
216 {
217 if (a <= 0.0 || b <= 0.0) {
218 string msg = string.Format ("Beta parameters must be positive. Received {0} and {1}.", a, b);
219 throw new ArgumentOutOfRangeException (msg);
220 }
221
222 // There are more efficient methods for generating beta samples.
223 // However such methods are a little more efficient and much more complicated.
224 // For an explanation of why the following method works, see
225 // http://www.johndcook.com/distribution_chart.html#gamma_beta
226
227 double u = GetGamma (a, 1.0);
228 double v = GetGamma (b, 1.0);
229 return u / (u + v);
230 }
